```typescript
import Axios, { AxiosRequestConfig } from "axios";
import { supabaseBrowserClient } from "@/lib/supabase/client";

// baseURL MUST be empty string — OpenAPI paths include /api/v1/ prefix
// Next.js proxy routes /api/v1/* -> backend (avoids CORS and auth exposure)
const AXIOS_INSTANCE = Axios.create({ baseURL: "" });

export const customInstance = async <T>(
  config: AxiosRequestConfig,
  options?: AxiosRequestConfig
): Promise<T> => {
  const {
    data: { session },
  } = await supabaseBrowserClient.auth.getSession();
  const token = session?.access_token ?? null;

  const source = Axios.CancelToken.source();

  const promise = AXIOS_INSTANCE({
    ...config,
    ...options,
    headers: {
      ...(token ? { Authorization: `Bearer ${token}` } : {}),
      ...config.headers,
      ...options?.headers,
    },
    cancelToken: source.token,
  }).then(({ data }) => data);

  // @ts-ignore — Orval generated code calls promise.cancel() if TanStack Query cancels
  promise.cancel = () => {
    source.cancel("Query was cancelled by React Query");
  };

  return promise;
};

// Required exports — Orval generated files import these types
export type ErrorType<Error> = import("axios").AxiosError<Error>;
export type BodyType<BodyData> = BodyData;
```

Key design decisions (do NOT deviate):
- `baseURL: ""` — OpenAPI spec paths already include `/api/v1/` prefix. Any non-empty baseURL causes double-prefix `/api/v1/api/v1/...`
- `supabaseBrowserClient.auth.getSession()` — matches existing `lib/api/client.ts` auth pattern exactly
- Token injection is optional-auth (injects when available, does not throw when absent) — matches existing client.ts behavior where both authed and public endpoints use the same client
- `ErrorType` and `BodyType` exports are REQUIRED — Orval generated files import these types from the mutator module
- `CancelToken` for React Query cancellation support — Orval generated code calls `promise.cancel()` on query cancellation
- No `"use client"` directive needed — the file runs in client context via component imports; `@/lib/supabase/client` already has the directive

**1. Add `hooks.afterAllFilesWrite` to orval.config.ts:**

Add a `hooks` block to the `decodedApi` config object as a SIBLING of `input` and `output` (NOT inside `output`). The final structure must be:

```typescript
decodedApi: {
    input: { ... },    // existing — do not change
    output: { ... },   // existing — do not change
    hooks: {
      afterAllFilesWrite: "./node_modules/.bin/prettier --write",
    },
  },
```

CRITICAL placement rule: `hooks` is at the same level as `input` and `output` inside `decodedApi: { ... }`. Placing it inside `output` causes a TypeScript type error and the hook silently does not run.

Use the full path `"./node_modules/.bin/prettier --write"` (not plain `"prettier --write"`). RESEARCH.md documents that the plain string may fail if the binary is not in shell PATH when orval spawns the subprocess. The full path resolves from the package directory and avoids this pitfall. Orval appends the generated file paths automatically.

Do NOT add hooks to the `decodedApiZod` block — zod schemas do not need formatting (they are small and rarely read by humans).

Do NOT modify the existing `input` or `output` blocks — they are already correct from Phase 39.
