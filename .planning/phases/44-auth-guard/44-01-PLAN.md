2. Restructure `proxy()` to handle `/profile` BEFORE the admin logic. The profile branch only checks session (no admin role check). On no session, redirect to `/login?redirect={pathname}`:

```typescript
export async function proxy(req: NextRequest) {
  const res = NextResponse.next();
  const supabase = createSupabaseMiddlewareClient(req, res);
  const { data: { session } } = await supabase.auth.getSession();
  const pathname = req.nextUrl.pathname;

  // Profile: require session, redirect to login with return URL
  if (pathname === "/profile" || pathname.startsWith("/profile/")) {
    if (!session) {
      const loginUrl = new URL("/login", req.url);
      loginUrl.searchParams.set("redirect", pathname);
      return NextResponse.redirect(loginUrl);
    }
    return res;
  }

  // Admin: existing logic (session + admin role)
  if (!session) {
    return NextResponse.redirect(new URL("/", req.url));
  }
  const isAdmin = await checkIsAdmin(supabase, session.user.id);
  if (!isAdmin) {
    return NextResponse.redirect(new URL("/", req.url));
  }
  return res;
}
```

Key: Use `req.nextUrl.pathname` (path only, not full URL) for the redirect param to prevent open redirect. Keep all existing imports unchanged.
