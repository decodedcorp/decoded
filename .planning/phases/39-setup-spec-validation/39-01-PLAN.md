    2. Create `packages/web/orval.config.ts` with the following exact content:

    ```typescript
    import { defineConfig } from "orval";

    export default defineConfig({
      decodedApi: {
        input: {
          target: "../api-server/openapi.json",
          override: {
            transformer: (spec: Record<string, unknown>) => {
              // Remove multipart/binary upload endpoints that Orval cannot generate valid hooks for.
              // These 4 endpoints use FormData/binary — not representable as typed hooks.
              // CRITICAL: /api/v1/posts has both GET (list_posts — KEEP) and POST (create_post_without_solutions — REMOVE).
              // We must remove only the POST verb, not the entire path.
              const pathsToRemoveVerb: Record<string, string[]> = {
                "/api/v1/posts": ["post"],
                "/api/v1/posts/with-solutions": ["post"],
                "/api/v1/posts/upload": ["post"],
                "/api/v1/posts/analyze": ["post"],
              };

              const paths = spec.paths as Record<string, Record<string, unknown>> | undefined;
              if (paths) {
                for (const [path, verbs] of Object.entries(pathsToRemoveVerb)) {
                  if (paths[path]) {
                    for (const verb of verbs) {
                      delete paths[path][verb];
                    }
                    // If no verbs remain, remove the entire path entry
                    if (Object.keys(paths[path]).length === 0) {
                      delete paths[path];
                    }
                  }
                }
              }

              return spec;
            },
          },
        },
        output: {
          mode: "tags-split",
          target: "./lib/api/generated",
          schemas: "./lib/api/generated/models",
          client: "react-query",
          httpClient: "axios",
          override: {
            mutator: {
              path: "./lib/api/mutator/custom-instance.ts",
              name: "customInstance",
            },
            query: {
              useQuery: true,
              useMutation: true,
            },
          },
        },
      },
      decodedApiZod: {
        input: {
          target: "../api-server/openapi.json",
        },
        output: {
          client: "zod",
          target: "./lib/api/generated/zod",
          fileExtension: ".zod.ts",
        },
      },
    });
    ```

    KEY DESIGN DECISIONS in this config:
    - `input.target` is `"../api-server/openapi.json"` — relative from `packages/web/` (where orval.config.ts lives) to `packages/api-server/openapi.json`. Orval resolves paths relative to the config file location. This file will be created in Plan 02 by downloading the live spec from the running backend.
    - `mode: "tags-split"` — generates one file per OpenAPI tag (posts.ts, feed.ts, admin.ts, etc.)
    - `httpClient: "axios"` — MUST be explicit because Orval 8 defaults to fetch
    - `mutator.path` references `./lib/api/mutator/custom-instance.ts` — this file does NOT exist yet, it will be created in Phase 40 (INFRA-02). The config is a skeleton.
    - The transformer removes POST verbs from 4 multipart paths while preserving GET verbs on shared paths like `/api/v1/posts`
    - Two config blocks: `decodedApi` for react-query hooks, `decodedApiZod` for standalone Zod schemas
    - The `decodedApiZod` input does NOT need the transformer because Zod schemas for request bodies of excluded endpoints are harmless (they just won't be used)

    IMPORTANT — path resolution note (from RESEARCH.md correction): The RESEARCH.md skeleton shows `./packages/api-server/openapi.json` which assumes monorepo-root invocation. Since orval.config.ts lives in `packages/web/` and Orval resolves `input.target` relative to the config file, the correct path is `../api-server/openapi.json`. The output paths (`./lib/api/generated`, `./lib/api/mutator/custom-instance.ts`) are already correct relative to `packages/web/`.
